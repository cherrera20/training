\section{Objetivos}

\begin{frame}  
  \begin{block}{Objetivos Principales}
    \begin{itemize}
    \item Entender los fundamentos del paradigma de la Programación Funcional
    \item Identificar y erradicar los \alert{efectos de lado} de un programa
    \item Aplicar la separación de aspectos: efectos e interpretación
    \end{itemize}
  \end{block}
  \begin{block}{Y de forma indirecta...}
    \begin{itemize}
    \item Aprender los fundamentos de la programación en \emph{Scala}
    \item Poder desenvolverse en el ecosistema Scala (REPL, SBT, ScalaTest...)
    \item Conocer las nociones básicas del framework web \emph{Play}
    \end{itemize}
  \end{block}
\end{frame}
